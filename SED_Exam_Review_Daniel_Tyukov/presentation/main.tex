\begin{frame}{Input of A1: the pick-up coil}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      The coil output is proportional to the \emph{rate of change} of magnetic flux,
      it \textbf{differentiates}:
      \begin{resultbox}
        $\displaystyle v_{coil}(t) = -N\,\frac{d\Phi}{dt}\;\propto\;\frac{dB}{dt}$
      \end{resultbox}
      \vskip0.3em
      \begin{itemize}
        \item Current in the loop $\to$ magnetic field $\to$ coil voltage.
        \item Open-circuit sensitivity \SI{-59.4}{dBV/(A/m)} at \SI{1}{kHz}.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{whybox}
      If the coil differentiates, A1 must \textbf{integrate}, so the cascade is flat across
      the audio band. That single fact drives the whole topology.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Output of A1: source-select switch + ADC}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item \textbf{Source-select switch} (mic / hearing loop): needs $V_{mic}=V_{loop}$ at
          $f_{ref}=\SI{1}{kHz}$.
        \item \textbf{ADC input} is a voltage amplifier: $Z_{in}\to\infty$, draws no current.
        \item Only loading left = parasitic \hl{C_L=\SI{10}{pF}}.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{block}{Consequence}
        A1 is a \textbf{voltage-to-voltage} amplifier driving a capacitor, not a power stage.
      \end{block}
      \begin{whybox}
      Matching $V_{mic}=V_{loop}$ at $f_{ref}$ fixes the required gain magnitude.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{How should A1 work?}
  \begin{enumerate}
    \item A1 is a \textbf{voltage-to-voltage amplifier}.
    \item Its transfer is an \textbf{integrator} (cancels the coil's differentiation):
      \begin{resultbox}
        $\displaystyle H(s)=\frac{V_o}{V_i}=\frac{1/\tau_i}{s}=\frac{\hl{62.4\times10^{3}}}{s}\ \si{V/V},
        \qquad \tau_i=\SI{16.06}{\micro s}$
      \end{resultbox}
    \item Gain set so $V_{mic}=V_{coil}$ at $f_{ref}$.
    \item \textbf{Noise:} weighted output noise $\le$ microphone floor ($\le\SI{30}{dBSPL}$).
  \end{enumerate}
  \begin{antoonbox}
  The order matters: transfer first, \emph{then} noise. And it stays a V--V amplifier
  throughout, no silent switch to a transimpedance amplifier.
  \end{antoonbox}
\end{frame}

\section{Transfer function \& feedback configuration}
% =================================================================

\begin{frame}{First generate the options, \emph{then} choose}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Option A, voltage-to-voltage} (chosen)
      \begin{itemize}
        \item $Z_1=$ capacitor, $Z_2=$ resistor.
        \item Coil $R_s$ enters only the \emph{noise}.
      \end{itemize}
      \vskip0.3em
      \textbf{Option B, current--voltage (transimpedance)}
      \begin{itemize}
        \item $Z_2=$ inductor integrates.
        \item Coil $R_s$ enters the \emph{transfer}; large inductor.
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{antoonbox}
      The prereview lesson: \textbf{pick one and stay consistent}. Mixing a V--V amplifier with
      a transimpedance amplifier between slides makes the plots meaningless.
      \end{antoonbox}
      \begin{block}{Decision}
        \textbf{Option A: voltage-to-voltage integrator.} Practical components, and $R_s$ stays out
        of the transfer.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Why feedback? The asymptotic-gain model}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      Feedback fixes the transfer with passive components, so accuracy follows them, not the
      transistor.
      \begin{resultbox}
        $\displaystyle A_f=A_\infty\underbrace{\frac{-L}{1-L}}_{\text{servo}}
        +\underbrace{\frac{1}{1-L}}_{\text{direct}}$
      \end{resultbox}
      \begin{itemize}
        \item $A_\infty$: ideal (nullor) transfer $=1/(sR_fC_i)$.
        \item $L$: loop gain; servo $\to1$ when $|L|\gg1\Rightarrow A_f\to A_\infty$.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \centering
      \begin{tikzpicture}[>=Latex,line width=0.8pt,node distance=6mm]
        \node (sum)[draw,circle,inner sep=1pt]{$\Sigma$};
        \node (A)[draw,fill=tudgreen!20,right=9mm of sum,minimum height=7mm]{nullor};
        \node (out)[right=10mm of A]{$V_o$};
        \node (in)[left=8mm of sum]{$V_i$};
        \node (B)[draw,fill=tudyellow!30,below=7mm of A,minimum height=6mm]{$Z_1,Z_2$};
        \draw[->](in)--(sum);\draw[->](sum)--(A);\draw[->](A)--(out);
        \draw[->]($(A)!0.5!(out)$)|-(B);\draw[->](B)-|(sum);
      \end{tikzpicture}\\[0.3em]
      \begin{antoonbox}
      In the review, \emph{explain every curve} of this model, gain, asymptotic, loop gain,
      servo, direct.
      \end{antoonbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Sizing the feedback network}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      Bound $R_f$ from two budgets:
      \begin{itemize}
        \item \textbf{Noise} ($B_{n1}=0.4$) $\Rightarrow$ upper bound $R_{f,max}=\SI{7.5}{k\ohm}$.
        \item \textbf{Power} ($B_{i1}=0.1$) $\Rightarrow$ lower bound $R_{f,min}=\SI{1.4}{k\ohm}$.
      \end{itemize}
      \[ C_i=\frac{\tau_i}{R_f},\qquad R_b=\frac{1}{2\pi C_i f_{min}}. \]
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{resultbox}
        $R_f=\hl{\SI{3.24}{k\ohm}}$\\[2pt]
        $C_i=\hl{\SI{4.96}{nF}}\qquad R_b=\hl{\SI{53.5}{k\ohm}}$
      \end{resultbox}
      \begin{whybox}
      $R_b\parallel C_i$ sets the low corner at $f_{min}=\SI{600}{Hz}$ and gives the DC bias a
      path, one resistor, two justified jobs.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\section{First stage: noise design}
% =================================================================

\begin{frame}{The first stage sets the input-referred noise}
  \begin{columns}[T]
    \begin{column}{0.44\textwidth}
      \includegraphics[width=\linewidth]{noisyNullorN}\\
      \src{Noisy nullor: ideal nullor + the input device's $v_n,i_n$}
    \end{column}
    \begin{column}{0.56\textwidth}
      Model the controller as a \textbf{noisy nullor}.
      \begin{itemize}
        \item Later stages' noise is divided by the first-stage gain $\Rightarrow$ negligible.
        \item So the whole budget is spent on \emph{stage 1}.
      \end{itemize}
      \begin{whybox}
      This is why the first stage is designed \emph{from the noise spec}, before anything else.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{The knobs: $g_m$, $c_{iss}$, inversion}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      Input-referred thermal noise of a CS stage:
      \begin{resultbox}$\displaystyle \overline{v_n^2}=\frac{4kT\gamma}{g_m}$\end{resultbox}
      \begin{itemize}
        \item More $g_m$ $\Rightarrow$ less noise (more current for less voltage).
        \item Wider device $\Rightarrow$ more $c_{iss}$ $\Rightarrow$ more input loading.
        \item \textbf{Weak inversion} gives the largest $g_m/I_D$.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{block}{The trade-off}
        Minimise $g_m\cdot c_{iss}$: enough $g_m$ for the noise budget, lowest $c_{iss}$ for the
        source.
      \end{block}
      \begin{antoonbox}
      ``Replace the how with the why'': weak inversion is chosen \emph{because} it maximises
      transconductance efficiency.
      \end{antoonbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Which stage? CS vs CG vs CD}
  \centering
  \renewcommand{\arraystretch}{1.15}
  \begin{tabular}{@{}l c c l@{}}
    \toprule
    & \textbf{Gain} & \textbf{Chain (ABCD)} & \textbf{Role} \\ \midrule
    \textbf{CS} & high & \textbf{smallest} entries & most \emph{nullor-like} \\
    CG & $\approx1$ (current) & $-$ & current follower \\
    CD & $\approx1$ (voltage) & $-$ & voltage follower \\
    \bottomrule
  \end{tabular}
  \vfill
  \begin{columns}[T]
    \begin{column}{0.62\textwidth}
      \begin{itemize}
        \item The CS chain matrix has the smallest elements $\Rightarrow$ most nullor-like, so
          later stages barely add noise.
        \item CG / CD are followers, no voltage gain to suppress later noise.
      \end{itemize}
    \end{column}
    \begin{column}{0.38\textwidth}
      \begin{resultbox}First stage $=$ \textbf{common-source}, in saturation.\end{resultbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{From budget to device parameters}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{enumerate}
        \item Find the output noise spectrum $S_o(f)$.
        \item Integrate (DIN-A weighted) $\to$ variance $\sigma_o^2=(B_{n2}V_{o,n})^2$.
        \item Express $g_m$ as a function of $c_{iss}$.
        \item Lowest feasible $c_{iss}\to g_m,I_D$, then $W,L$ via the \textbf{EKV} model.
      \end{enumerate}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{resultbox}
        PMOS input ($g_m\!\cdot\!c_{iss}$ optimum):\\[2pt]
        $g_m=\hl{\SI{73}{\micro S}}$,\ $W=\SI{19.1}{\micro m}$\\[2pt]
        $L=\SI{0.72}{\micro m}$,\ $I_D=\SI{8.1}{\micro A}$
      \end{resultbox}
      \begin{whybox}
      SLiCAP gives the symbolic spectrum; EKV turns the budget into $W,L,I_D$; ngspice verifies.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Is a single stage enough?}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \includegraphics[width=\linewidth]{single_stage_loopgain}\\
      \src{Loop gain $L$ stays below \SI{0}{dB}.}
    \end{column}
    \begin{column}{0.5\textwidth}
      A single CS has only 3 usable terminals $\Rightarrow$ a \textbf{differential pair}
      ($\approx4\times$ current/area).
      \begin{itemize}
        \item Current drive into $C_L$: \textcolor{tudgreen}{\checkmark}
        \item Voltage headroom on \SI{0.9}{V}: \textcolor{tudgreen}{\checkmark}
        \item Loop gain $|L_{DC}|<1$ across the band: \textcolor{tudorange}{\ding{55}}
      \end{itemize}
      \begin{resultbox}
        $|L|<1\Rightarrow$ servo cannot enforce the transfer $\Rightarrow$ \textbf{poor accuracy}.
        Go to a \textbf{dual stage}.
      \end{resultbox}
    \end{column}
  \end{columns}
\end{frame}

% =================================================================
\section{Dual stage (PMOS pair + NMOS CS)}
% =================================================================

\begin{frame}{Why a dual stage?}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      Five competing costs: \emph{noise, power, accuracy, stability, area}.
      \begin{itemize}
        \item The \textbf{first stage dominates noise}.
        \item The \textbf{second stage dominates loop gain} (accuracy).
      \end{itemize}
      Split the job, optimise each stage for one thing.
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{resultbox}
        Stage 1: \textbf{PMOS differential pair}, low noise.\\[3pt]
        Stage 2: \textbf{NMOS common-source}, high loop gain.
      \end{resultbox}
      \begin{whybox}
      One stage could not give both low noise and enough loop gain; two stages can.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Suggested topology}
  \centering
  \includegraphics[width=0.62\linewidth]{dualstage_colored}\\[0.3em]
  {\footnotesize
   \cbox{tudgray}~coil \ \ \cbox{tudcyan}~1st stage: PMOS pair \ \
   \cbox{tudyellow}~feedback \ \ \cbox{tudpink}~2nd stage: NMOS CS \ \ \cbox{tudgreen}~ADC load}\\[0.2em]
  \src{EKV schematic, \texttt{.source V1}, \texttt{.detector V\_out}, \texttt{.lgref Gm\_M1\_X2}}
\end{frame}

\begin{frame}{First stage: differential pair vs single CS}
  \centering
  \renewcommand{\arraystretch}{1.15}
  \begin{tabular}{@{}p{0.46\textwidth} p{0.46\textwidth}@{}}
    \toprule
    \textbf{Differential pair} & \textbf{Single-ended CS} \\ \midrule
    \textcolor{tudgreen}{+} lower noise & \textcolor{tudgreen}{+} less area \\
    \textcolor{tudgreen}{+} high CMRR & \textcolor{tudgreen}{+} lower power \\
    \textcolor{tudgreen}{+} matches the coil's 2-terminal input & \textcolor{tudgreen}{+} simple \\
    \textcolor{tudorange}{--} more area & \textcolor{tudorange}{--} allows CM interference \\
    \textcolor{tudorange}{--} higher complexity & \textcolor{tudorange}{--} higher noise \\
    \bottomrule
  \end{tabular}
  \vfill
  \begin{resultbox}
  The coil is a floating two-terminal source $\Rightarrow$ the \textbf{differential pair} wins on
  noise and common-mode rejection.
  \end{resultbox}
\end{frame}

\begin{frame}{Second stage: NMOS common-source}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      Chosen for what the first stage is \emph{not} optimised for:
      \begin{itemize}
        \item high loop gain $\Rightarrow$ accuracy;
        \item high current-drive into $C_L$;
        \item low power, small area, simple.
      \end{itemize}
      Minimum length for speed; width traded for $g_m/I_D$.
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{resultbox}
        NMOS CS output stage:\\[2pt]
        $W=\SI{10}{\micro m}$,\ $L=\SI{0.18}{\micro m}$\\[2pt]
        $I_D=\hl{\SI{16.5}{\micro A}}$
      \end{resultbox}
      \begin{whybox}
      Drive current $I_D\approx V_{i,pp}/2R_f$ must charge $C_L$ at the full-power frequency.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Dual-stage design parameters}
  \centering
  \includegraphics[width=0.8\linewidth]{dualStageSimple_colored}\\[0.2em]
  {\footnotesize
   \cbox{tudgray}~coil \ \ \cbox{tudcyan}~1st stage ($g_{m1},g_{o1}$) \ \
   \cbox{tudpink}~2nd stage ($g_{m2}$) \ \ \cbox{tudyellow}~feedback ($R_f$) \ \
   \cbox{tudgreen}~ADC load}\\[0.35em]
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \footnotesize
      \textbf{$g_{m1}$} first-stage transconductance, set by the \emph{noise} budget.\\[2pt]
      \textbf{$g_{m2}$} second-stage transconductance, the extra \emph{loop-gain} factor.\\[2pt]
      \textbf{$R_f$} feedback resistor, set by the \emph{transfer}.\\[2pt]
      \textbf{$g_{o1}$} first-stage output conductance (its output resistance).
    \end{column}
    \begin{column}{0.5\textwidth}
      \resizebox{\linewidth}{!}{$\displaystyle L_{DC}=\frac{R_f g_{m1} g_{m2}(-R_s-R_t)}{R_sg_{o1}+R_tg_{o1}}=-\frac{R_f g_{m1}g_{m2}}{g_{o1}}$}
      \vskip0.4em
      \begin{resultbox}$|L_{DC}|\approx\SI{45}{dB}$, enough for accuracy.\end{resultbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Asymptotic-gain model: every curve explained}
  \begin{columns}[T]
    \begin{column}{0.6\textwidth}
      \includegraphics[width=\linewidth]{fb_decomposition}
    \end{column}
    \begin{column}{0.4\textwidth}
      \footnotesize
      \begin{itemize}
        \item \textbf{\textcolor{tudblue}{gain}}: the closed-loop transfer we deliver.
        \item \textbf{\textcolor{tudcyan}{asymptotic} $A_\infty$}: ideal nullor target $1/sR_fC_i$.
        \item \textbf{\textcolor{tudorange}{loop gain} $L$}: $\approx\SI{45}{dB}$ at DC, budget for accuracy.
        \item \textbf{\textcolor{tudgreen}{servo} $L/(1{-}L)$}: $\to1$ while $|L|\gg1$, so gain $\to A_\infty$.
        \item \textbf{\textcolor{tudpink}{direct} $1/(1{-}L)$}: feed-through; negligible.
      \end{itemize}
      \begin{antoonbox}
      Gain follows $A_\infty$ exactly where the servo $\approx1$; it departs once $|L|$ falls.
      \end{antoonbox}
    \end{column}
  \end{columns}
\end{frame}

\section{Frequency compensation (MFM)}
% =================================================================

\begin{frame}{Target: maximally-flat magnitude (MFM)}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      A pole pair with $Q>\tfrac{1}{\sqrt2}$ peaks and rings. \textbf{MFM} (Butterworth) sets the
      dominant pair to
      \[ Q=\tfrac{1}{\sqrt2}\approx0.707. \]
      \begin{itemize}
        \item No peaking, fastest flat response.
        \item Comfortable phase margin.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{block}{How to move a pole pair?}
        Add a \emph{zero} to the loop gain near the offending poles, it injects phase lead and
        lowers the effective $Q$.
      \end{block}
      \begin{whybox}
      We shape \emph{stability}, not just gain, a Bode plot is the $0$-axis slice of the
      Nyquist plot.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{The phantom-zero method}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      Put $R_{phz}$ in series with the load capacitor: a zero in the \emph{loop gain} with no real
      pole in the signal path.
      \begin{resultbox}$\displaystyle z=-\frac{1}{R_{phz}C_L}$\end{resultbox}
      \textbf{Why this one?}
      \begin{itemize}
        \item no extra noise source in the signal path;
        \item negligible power and area;
        \item no added energy storage (no compensation cap).
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \centering
      \begin{tikzpicture}[scale=0.8,transform shape,line width=0.8pt]
        \node[draw,fill=tudpink!25,minimum height=10mm,minimum width=15mm] (amp) at (0,0){NMOS CS};
        \draw (amp.east)--(1.6,0) coordinate(a);
        \draw (a) to[R,l=$R_{phz}$] (4.2,0) coordinate(out);
        \draw (out)--(4.9,0) node[right]{\textbf{out}};
        \draw (out) to[C,l_=$C_L$] (4.2,-1.9) node[ground]{};
        \draw (a)--(a|-0,1.5) coordinate(t) to[C,l=$C_i$] (-1.0,1.5);
        \node[above right,font=\scriptsize] at (t){feedback tap};
      \end{tikzpicture}\\[0.3em]
      \src{$R_{phz}$ in series; feedback tapped \emph{before} it.}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Compensation result}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \includegraphics[width=\linewidth]{compensation_mfm}
    \end{column}

      \textbf{The structured path, end to end:}
      \begin{enumerate}\footnotesize
        \item Specification \& budget from the environment.

        \item Build the model up, block by block.
        \item Noise budget $\to$ first-stage $g_m,c_{iss}\to W,L,I_D$.

        \item PMOS pair (noise) + NMOS CS (loop gain).

      \end{enumerate}

      component justified, the how replaced by the why.
      \end{antoonbox}

        \footnotesize $H(s)=\dfrac{62.4\times10^{3}}{s}$, noise $\le\SI{10.6}{\micro V}$, stable, MFM.

\end{frame}
