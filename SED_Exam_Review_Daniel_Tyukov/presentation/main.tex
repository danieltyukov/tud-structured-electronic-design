\begin{frame}{Representation 1 --- the pick-up coil (a \emph{differentiator})}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      A current in the loop makes a magnetic field; the coil sees a voltage proportional
      to the \emph{rate of change} of flux:
      \begin{resultbox}
        $\displaystyle v_{coil}(t) = -N\,\frac{d\Phi}{dt}\;\propto\; \frac{dB}{dt}$
      \end{resultbox}
      \vskip0.4em
      Small-signal model of the coil + cable:
      \begin{itemize}
        \item series $L_s=\SI{120}{mH}$, $R_s=\SI{875}{\ohm}$
        \item parasitic $C_s\approx\SI{9.4}{pF}$
        \item termination $R_t\approx\SI{80}{k\ohm}$
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \centering
      \begin{circuitikz}[scale=0.78,transform shape,line width=0.8pt]
        \draw (0,0) to[sV,l=$v_{emf}$] (0,2.6);
        \draw (0,2.6) to[L,l=$L_s$] (2,2.6) to[R,l=$R_s$] (4,2.6) coordinate(in);
        \draw (in) to[short] (5.2,2.6) node[right]{\textbf{in}};
        \draw (in) to[C,l_=$C_s$] (in|-0,0);
        \draw (5.2,2.6) to[R,l=$R_t$] (5.2,0);
        \draw (0,0) -- (5.2,0) node[ground]{};
      \end{circuitikz}\\[0.3em]
      \src{Pick-up coil two-port}
      \begin{whybox}
      An \textbf{integrator} A1 undoes this differentiation $\Rightarrow$ flat audio response.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Representation 2 --- the termination resistor}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      \includegraphics[width=\linewidth]{coil_damping}
    \end{column}
    \begin{column}{0.44\textwidth}
      $L_s$ and $C_s$ resonate at
      \[ f_{res}=\frac{1}{2\pi\sqrt{L_s C_s}}\approx\SI{150}{kHz}. \]
      Undamped, this peak rings and overloads the input.
      \begin{itemize}
        \item Add $R_t$ across the input.
        \item Pick $R_t$ for \emph{critical} damping, $Q=\tfrac{1}{\sqrt2}$.
      \end{itemize}
      \begin{antoonbox}
      ``If you add a component, have a \emph{reason}.'' $R_t$ exists only to damp the LC
      resonance --- and it is then a noise source we must budget.
      \end{antoonbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Representation 3 --- the ideal amplifier (nullor)}
  \begin{columns}[T]
    \begin{column}{0.53\textwidth}
      I model the controller as a \textbf{nullor}: the ideal element with infinite gain.
      \begin{itemize}
        \item Input (nullator): $v=0,\ i=0$.
        \item Output (norator): $v,i$ free.
      \end{itemize}
      With a nullor the transfer is set \emph{only} by the passive feedback:
      \begin{resultbox}
        $\displaystyle A_\infty=\frac{1}{s\,R_f C_i}=\frac{62.4\times10^3}{s}$
      \end{resultbox}
    \end{column}
    \begin{column}{0.47\textwidth}
      \centering
      \begin{tikzpicture}[line width=0.9pt]
        \fill[tudcyan!20,draw=tudblue] (0,0)--(2.5,1.0)--(0,2.0)--cycle;
        \node at (0.78,1.0) {\large $\infty$};
        \node[below=1pt,font=\small] at (1.2,0) {nullor};
        \draw (-0.7,1.55)--(0,1.55) node[left=16pt]{\small$-$};
        \draw (-0.7,0.55)--(0,0.55) node[left=16pt]{\small$+$};
        \draw (2.5,1.0)--(3.2,1.0) node[right]{\small out};
      \end{tikzpicture}\\[0.5em]
      \begin{whybox}
      The nullor separates the \emph{ideal target} from the implementation. Real transistors
      come later.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Representation 4 --- the output (switch, ADC, $C_L$)}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \begin{itemize}
        \item \textbf{Source-select switch} (microphone / hearing loop).
          \begin{itemize}\item requires $V_{mic}=V_{loop}$ at the reference frequency $f_{ref}=\SI{1}{kHz}$.\end{itemize}
        \item \textbf{ADC input} behaves as a voltage amplifier: $Z_{in}\to\infty$, draws no
          current.
        \item Only loading left = parasitic \hl{C_L=\SI{10}{pF}}.
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{block}{Consequence}
        A1 is a \textbf{voltage--voltage} amplifier driving a capacitive load --- not a
        power stage.
      \end{block}
      \begin{whybox}
      Matching $V_{mic}=V_{loop}$ at $f_{ref}$ is what fixes the required gain magnitude.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Representation 5 --- the noise sources}
  \begin{columns}[T]
    \begin{column}{0.47\textwidth}
      \includegraphics[width=\linewidth]{noise_spectrum}
    \end{column}
    \begin{column}{0.53\textwidth}
      Each resistor adds thermal noise $\overline{v^2}=4kTR$; the input device adds $v_n,i_n$
      (the \emph{noisy nullor}).
      \begin{itemize}
        \item Budget $B_{n1}=0.4$ to $R_s$, $R_t$, $R_f$.
        \item The rest goes to the input transistor $\Rightarrow$ this sets $g_m$.
        \item Output-noise RMS $\approx\SI{7.5}{\micro V}$ (1\,Hz--100\,kHz), within the
          \SI{10.6}{\micro V} DIN-A budget. \textcolor{tudgreen}{\checkmark}
      \end{itemize}
      \begin{antoonbox}
      Every noise source is traced to the budget --- nothing is added without accounting for it.
      \end{antoonbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{How should A1 work?}
  \begin{enumerate}
    \item A1 is a \textbf{voltage--voltage amplifier}.
    \item Its transfer is an \textbf{integrator} (cancels the coil's differentiation):
      \begin{resultbox}
        $\displaystyle H(s)=\frac{V_o}{V_i}=\frac{1/\tau_i}{s}
        =\frac{\hl{62.4\times10^{3}}}{s}\ \si{V/V},\qquad \tau_i=\SI{16.06}{\micro s}$
      \end{resultbox}
    \item Gain magnitude set so $V_{mic}=V_{coil}$ at $f_{ref}$.
    \item \textbf{Noise:} weighted output noise must not exceed the microphone's floor
      ($\le\SI{30}{dBSPL}$).
  \end{enumerate}
  \begin{antoonbox}
  Note the \emph{order}: transfer first, \emph{then} noise. Keep it a V--V amplifier
  throughout --- do not silently switch to a transimpedance amplifier.
  \end{antoonbox}
\end{frame}

\begin{frame}{Noise is judged by ear: DIN-A weighting}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      \includegraphics[width=\linewidth]{din_a_weighting}
    \end{column}
    \begin{column}{0.44\textwidth}
      The ear is not flat. The output noise is weighted by the standard \textbf{DIN-A}
      curve before integrating to an RMS value:
      \[ \sigma_o^2=\int_0^\infty \big|A(f)\big|^2 S_o(f)\,df. \]
      \begin{itemize}
        \item $0$ dB at \SI{1}{kHz}.
        \item Strong roll-off below \SI{600}{Hz} / above a few kHz.
      \end{itemize}
      \begin{whybox}
      Weighting tightens the budget exactly where hearing is sensitive --- so the budget is
      perceptually meaningful.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\section{Transfer function \& feedback configuration}
% =================================================================

\begin{frame}{First generate the options --- \emph{then} choose}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Option A --- voltage--voltage}\\ (integrator in feedback)
      \begin{itemize}
        \item $Z_1=$ capacitor, $Z_2=$ resistor.
        \item Coil seen as a voltage source.
        \item Coil $R_s$ only enters \emph{noise}.
      \end{itemize}
      \vskip0.4em
      \textbf{Option B --- current--voltage}\\ (transimpedance, inductor integrates)
      \begin{itemize}
        \item $Z_2=$ inductor, $Z_1=$ resistor.
        \item Coil $R_s$ now enters the \emph{transfer}.
        \item Large inductor $\Rightarrow$ area / noise.
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{antoonbox}
      The prereview lesson: \textbf{pick one and stay consistent.} Mixing a V--V amplifier
      on one slide with a transimpedance amplifier on the next makes the plots meaningless.
      \end{antoonbox}
      \vskip0.5em
      \begin{block}{Decision}
        Choose \textbf{Option A: voltage--voltage integrator}.
        Practical components, and $R_s$ stays out of the transfer.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Why feedback? The asymptotic-gain model}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      Feedback fixes the transfer with a passive network, so accuracy follows passive
      components --- not the transistor.
      \begin{resultbox}
        $\displaystyle A_f \;=\; A_\infty\underbrace{\frac{-L}{1-L}}_{\text{servo}}
        \;+\;\underbrace{\frac{1}{1-L}}_{\text{direct}}$
      \end{resultbox}
      \begin{itemize}
        \item $A_\infty$: ideal (nullor) transfer = $1/(sR_fC_i)$.
        \item $L$: loop gain (return ratio).
        \item servo $\to 1$ when $|L|\gg1$ $\Rightarrow$ $A_f\to A_\infty$.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \centering
      \begin{tikzpicture}[>=Latex,line width=0.8pt,node distance=6mm]
        \node (sum) [draw,circle,inner sep=1.2pt] {$\Sigma$};
        \node (A) [draw,fill=tudcyan!20,right=10mm of sum,minimum height=7mm] {nullor};
        \node (out)[right=12mm of A] {$V_o$};
        \node (in) [left=10mm of sum] {$V_i$};
        \node (B) [draw,fill=tudyellow!30,below=8mm of A,minimum height=6mm] {$Z_1,Z_2$};
        \draw[->] (in)--(sum); \draw[->] (sum)--(A); \draw[->] (A)--(out);
        \draw[->] ($(A)!0.5!(out)$)|-(B); \draw[->] (B)-|(sum);
      \end{tikzpicture}\\[0.4em]
      \begin{antoonbox}
      In the review you must \emph{explain every curve} of this model --- gain, asymptotic,
      loop gain, servo, direct.
      \end{antoonbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Sizing the feedback network}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      Bound $R_f$ from two budgets:
      \begin{itemize}
        \item \textbf{Noise} budget ($B_{n1}=0.4$) $\Rightarrow$ \emph{upper} bound
          $R_{f,max}=\SI{7.5}{k\ohm}$.
        \item \textbf{Power} budget ($B_{i1}=0.1$) $\Rightarrow$ \emph{lower} bound
          $R_{f,min}=\SI{1.4}{k\ohm}$.
      \end{itemize}
      Then set $C_i$ from the integrator constant and add a bypass $R_b$ for the DC path:
      \[ C_i=\frac{\tau_i}{R_f},\qquad R_b=\frac{1}{2\pi C_i f_{min}}. \]
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{resultbox}
        $R_f=\hl{\SI{3.24}{k\ohm}}$\\[2pt]
        $C_i=\hl{\SI{4.96}{nF}}\qquad R_b=\hl{\SI{53.5}{k\ohm}}$
      \end{resultbox}
      \vskip0.4em
      \begin{whybox}
      $R_b\parallel C_i$ sets the lower corner at $f_{min}=\SI{600}{Hz}$ and gives the DC
      bias a path --- one resistor, two justified jobs.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\section{First stage: noise design}
% =================================================================

\begin{frame}{The first stage sets the input-referred noise}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \includegraphics[width=\linewidth]{noisyNullorN}\\
      \src{Noisy nullor: ideal nullor + the input device's $v_n,i_n$}
    \end{column}
    \begin{column}{0.54\textwidth}
      Replace the controller by a \textbf{noisy nullor}: an ideal nullor plus the input
      transistor's noise sources.
      \begin{itemize}
        \item Later stages add almost nothing (divided by the first-stage gain).
        \item So the whole noise budget is spent on \emph{stage 1}.
      \end{itemize}
      \begin{whybox}
      This is why we design the first stage \emph{from the noise spec} --- before sizing
      anything else.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{The knobs: $g_m$, $c_{iss}$, inversion}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      Input-referred thermal noise of a CS stage:
      \begin{resultbox}
        $\displaystyle \overline{v_n^2}=\frac{4kT\gamma}{g_m}\ \big[\si{V^2/Hz}\big]$
      \end{resultbox}
      \begin{itemize}
        \item More $g_m$ $\Rightarrow$ less noise (more current for less voltage).
        \item But wider device $\Rightarrow$ more $c_{iss}$ $\Rightarrow$ more input loading.
        \item \textbf{Weak inversion} gives the largest $g_m/I_D$.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{block}{The trade-off}
        Minimise the product $g_m\cdot c_{iss}$: enough $g_m$ for the noise budget, lowest
        $c_{iss}$ for the source.
      \end{block}
      \begin{antoonbox}
      ``Replace the how with the why'': we choose weak inversion \emph{because} it maximises
      transconductance efficiency for the noise budget.
      \end{antoonbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Which stage? CS vs CG vs CD}
  \centering
  \renewcommand{\arraystretch}{1.2}
  \begin{tabular}{@{}l c c l@{}}
    \toprule
    & \textbf{Gain} & \textbf{Chain (ABCD)} & \textbf{Role} \\
    \midrule
    \textbf{CS} & high & \textbf{smallest} entries & most \emph{nullor-like} \\
    CG & $\approx1$ (current) & --- & current follower \\
    CD & $\approx1$ (voltage) & --- & voltage follower \\
    \bottomrule
  \end{tabular}
  \vfill
  \begin{columns}[T]
    \begin{column}{0.62\textwidth}
      \begin{itemize}
        \item The CS chain matrix has the \emph{smallest} elements $\Rightarrow$ it behaves
          most like a nullor, so following stages barely add noise.
        \item CG and CD are \emph{followers} --- no voltage gain to suppress later noise.
      \end{itemize}
    \end{column}
    \begin{column}{0.38\textwidth}
      \begin{resultbox}
        First stage $=$ \textbf{common-source}, in saturation.
      \end{resultbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{From budget to device parameters}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{enumerate}
        \item Find the output noise spectrum $S_o(f)$.
        \item Integrate (DIN-A weighted) $\to$ variance $\sigma_o^2=(B_{n2}\,V_{o,n})^2$.
        \item Express $g_m$ as a function of $c_{iss}$.
        \item Lowest feasible $c_{iss}$ $\to$ $g_m,I_D$, then $W,L$ via the \textbf{EKV} model.
      \end{enumerate}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{resultbox}
        PMOS input ($g_m\!\cdot\!c_{iss}$ optimum):\\[2pt]
        $g_m=\hl{\SI{73}{\micro S}}$,\ \ $W=\SI{19.1}{\micro m}$\\[2pt]
        $L=\SI{0.72}{\micro m}$,\ \ $I_D=\SI{8.1}{\micro A}$
      \end{resultbox}
      \begin{whybox}
      SLiCAP gives the symbolic spectrum; the EKV model turns the budget directly into
      $W,L,I_D$. ngspice then verifies the numbers.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\section{Single stage \& loop-gain check}
% =================================================================

\begin{frame}{Is a single stage enough?}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \includegraphics[width=\linewidth]{single_stage_loopgain}\\
      \src{Single-stage feedback model --- the loop gain $L$ stays below \SI{0}{dB}.}
    \end{column}
    \begin{column}{0.5\textwidth}
      A single CS has only 3 usable terminals $\Rightarrow$ a \textbf{differential pair}
      ($\approx4\times$ current and area).
      \begin{itemize}
        \item Current-drive into $C_L$: \textcolor{tudgreen}{\checkmark}
        \item Voltage headroom on \SI{0.9}{V}: \textcolor{tudgreen}{\checkmark}
        \item Loop gain $|L_{DC}|<1$ across the band: \textcolor{tudorange}{\ding{55}}
      \end{itemize}
      \begin{block}{Verdict}
        $|L|<1$ $\Rightarrow$ the servo cannot enforce the transfer $\Rightarrow$ \textbf{poor
        accuracy}. Go to a \textbf{dual stage}.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% =================================================================
\section{Dual stage (PMOS pair + NMOS CS)}
% =================================================================

\begin{frame}{Why a dual stage?}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      The five costs trade against each other:\\
      \emph{noise, power, accuracy, stability, area}.
      \begin{itemize}
        \item The \textbf{first stage dominates noise}.
        \item The \textbf{second stage dominates loop gain} (accuracy).
      \end{itemize}
      Split the job: optimise each stage for one thing.
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{resultbox}
        Stage 1: \textbf{PMOS differential pair} --- low noise.\\[3pt]
        Stage 2: \textbf{NMOS common-source} --- high loop gain.
      \end{resultbox}
      \begin{whybox}
      Two stages give enough loop gain for accuracy while keeping the input noise low ---
      the single stage could not do both.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

  \begin{tabular}{@{}p{0.46\textwidth} p{0.46\textwidth}@{}}

    \textbf{Differential pair} & \textbf{Single-ended CS} \\
    \midrule
    \textcolor{tudgreen}{+} lower noise & \textcolor{tudgreen}{+} less area \\
    \textcolor{tudgreen}{+} high CMRR & \textcolor{tudgreen}{+} lower power \\
    \textcolor{tudgreen}{+} matches the coil's 2-terminal input & \textcolor{tudgreen}{+} simple \\
    \textcolor{tudorange}{--} more area & \textcolor{tudorange}{--} allows CM interference \\
    \textcolor{tudorange}{--} higher complexity & \textcolor{tudorange}{--} higher noise \\
    \bottomrule
  \end{tabular}
  \vfill
  \begin{resultbox}
  The coil is a floating two-terminal source $\Rightarrow$ the \textbf{differential pair}
  wins on noise and common-mode rejection.
  \end{resultbox}

\begin{frame}{Second stage: NMOS common-source}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      Chosen for what the first stage is \emph{not} optimised for:
      \begin{itemize}
        \item high loop gain $\Rightarrow$ accuracy;
        \item high current-drive into $C_L$;
        \item low power, small area, simple.
      \end{itemize}
      Minimum length $L=\SI{0.18}{\micro m}$ for speed; width traded for $g_m/I_D$.
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{resultbox}
        NMOS CS output stage:\\[2pt]
        $W=\SI{10}{\micro m}$,\ \ $L=\SI{0.18}{\micro m}$\\[2pt]
        $I_D=\hl{\SI{16.5}{\micro A}}$
      \end{resultbox}
      \begin{whybox}
      Drive current $I_D\approx V_{i,pp}/2R_f$ must charge $C_L$ at the full-power frequency.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

% =================================================================
\section{Frequency compensation (MFM)}
% =================================================================

\begin{frame}{Target: maximally-flat magnitude (MFM)}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      A pole pair with $Q>\tfrac{1}{\sqrt2}$ peaks and rings.
      \textbf{MFM} (Butterworth) sets the dominant closed-loop pair to
      \[ Q=\frac{1}{\sqrt2}\approx0.707. \]
      \begin{itemize}
        \item No peaking, fastest flat response.
        \item Comfortable phase margin.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{block}{How to move a pole pair?}
        Add a \emph{zero} to the loop gain near the offending poles --- it injects phase
        lead and lowers the effective $Q$.
      \end{block}
      \begin{whybox}
      We shape \emph{stability}, not just gain --- a Bode plot is the $0$-axis slice of the
      Nyquist plot.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{The phantom-zero method}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      Put a resistor $R_{phz}$ in series with the load capacitor. It creates a zero in the
      \emph{loop gain} without a real pole in the signal path:
      \begin{resultbox}
        $\displaystyle z=-\frac{1}{R_{phz}\,C_L}$
      \end{resultbox}
      \textbf{Why this one?}
      \begin{itemize}
        \item no extra noise source in the signal path;
        \item negligible power and area;
        \item no added energy-storage (no compensation cap).
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \centering
      \begin{circuitikz}[scale=0.8,transform shape,line width=0.8pt]
        \node[draw,fill=tudcyan!20,minimum height=10mm,minimum width=15mm] (amp) at (0,0) {NMOS CS};
        \draw (amp.east) -- (1.6,0) coordinate(a);
        \draw (a) to[R,l=$R_{phz}$] (4.2,0) coordinate(out);
        \draw (out) -- (4.9,0) node[right]{\textbf{out}};
        \draw (out) to[C,l_=$C_L$] (4.2,-1.9) node[ground]{};
        \draw (a) -- (a|-0,1.5) coordinate(t) to[C,l=$C_i$] (-1.0,1.5);
        \draw (t) node[above right,font=\scriptsize]{feedback tap};
      \end{circuitikz}\\[0.4em]
      \src{$R_{phz}$ in series with the output; feedback is taken \emph{before} it,
      so the zero appears only in the loop gain.}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Compensation result}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \includegraphics[width=\linewidth]{compensation_mfm}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{resultbox}
        $R_{phz}\approx\hl{\SI{0.70}{k\ohm}}$\\[3pt]
        $Q:\ 5.4 \;\longrightarrow\; \hl{0.71}$ \;(MFM)
      \end{resultbox}
      \begin{itemize}
        \item Peak removed; passband untouched.
        \item All poles remain in the left half-plane $\Rightarrow$ stable.
        \item Cost: one resistor.
      \end{itemize}
      \begin{whybox}
      The phantom zero buys maximally-flat magnitude for essentially zero noise/power penalty.
      \end{whybox}
    \end{column}
  \end{columns}
\end{frame}
